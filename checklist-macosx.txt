o Build CGAL:

  cd CGAL-3.5
  source runcmake.sh
  make -j4
  make install

o Patch OpenCSG

  cd OpenCSG-1.1.1
  patch -p1 < ../openscad/OpenCSG-MacOSX-port.patch
  patch -p1 < ../openscad/OpenCSG-Reset-Hack.patch      # Only if MDI

o Build OpenCSG

  qmake -recursive
  make
  
o Build OpenSCAD

# If MDI: edit openscad.pro and set ENABLE_MDI=1
  cd openscad
  qmake
  make -j4


o Deploy

  ./release-macosx.sh
