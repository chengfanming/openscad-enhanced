o Prerequisites (MacPorts):
  - NB! CGAL requires gcc >= 4.2
  - eigen
  - boost
  - gmp
  - mpfr
  - Qt4

o Build CGAL:

  tar xzf CGAL-3.5.1.tar.gz
  cd CGAL-3.5.1
  patch -p1 < ../openscad/patches/CGAL-OGL-Tess-Combine-Fix.patch
  cmake -DCMAKE_INSTALL_PREFIX=$PWD/../install -DBUILD_SHARED_LIBS=FALSE
  make -j4
  make install

o Patch OpenCSG

  cd OpenCSG-1.3.0
  patch -p1 < ../openscad/patches/OpenCSG-1.3.0-MacOSX-port.patch

o Build OpenCSG

  qmake -recursive
  make
  
o Build and Deploy OpenSCAD

# Update VERSION in publish-macosx.sh
  cd openscad
  ./scripts/publish-macosx.sh
